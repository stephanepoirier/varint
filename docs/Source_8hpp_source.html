35  block_size_ = block_size > 0 ? block_size : left_;

38  // Return the number of bytes left to read from the source
39  __inline__ size_t Available() const { return left_; }

41  // To minimize the amount of data copying that needs to be done
42  // return a pointer and size to internal buffer instead of copying to caller buffer.
43  // The returned region is valid until the next call to Skip()
44  // The caller is responsible for ensuring that it only reads uptop bytes_in_buffer
45  __inline__ const uint8* Peek(size_t* bytes_in_buffer) {
46  *bytes_in_buffer = left_;
47  return ptr_;
48  }

50  // Obtains a chunk of data from the stream.
51  // If the returned value is false, there is no more data to return or an error occurred.
52  // Otherwise, "size" points to the actual number of bytes read and "data" points to a pointer to a buffer containing these bytes.
53  __inline__ bool Next(const void** data, size_t* size) {
54  if (!left_) return false;
55 
56  if (left_ <= block_size_) {
57  Skip(left_);
58  } else {
59  Skip(block_size_);
60  }
61 
62  *data = Peek(size);
63  return true;
64  }
65 
66  // Skip the next n bytes
67  // REQUIRES: Available() >= n
68  // Returns false if the end of the stream is reached or some input error occurred.
69  // In the end-of-stream case, the stream is advanced to the end of the stream
70  // (so ByteCount() will return the total size of the stream).
71  __inline__ bool Skip(size_t count) {
72  //if (count > left_) return false;
73  left_ -= count;
74  ptr_ += count;
75  return true;
76  }
